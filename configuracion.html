4. Configuración (XML)

Este es el "mapa" que le dice a Hibernate dónde está tu base de datos. Crea el archivo hibernate.cfg.xml dentro de la carpeta src.

hibernate.cfg.xml 
<?xml version="1.0" encoding="UTF-8"?>
<!-- Si Eclipse da error de descarga, borra las líneas de DOCTYPE -->
<!DOCTYPE hibernate-configuration PUBLIC
        "-//Hibernate/Hibernate Configuration DTD 3.0//EN"
        "http://www.hibernate.org/dtd/hibernate-configuration-3.0.dtd">
        
<hibernate-configuration>
    <session-factory>
        <!-- 1. CONEXIÓN (Modificar estos datos) -->
        <property name="connection.driver_class">com.mysql.cj.jdbc.Driver</property>
        <property name="connection.url">jdbc:mysql://localhost:3306/TU_BASE_DE_DATOS</property>
        <property name="connection.username">root</property>
        <property name="connection.password">TU_CONTRASEÑA</property>

        <!-- 2. CONFIGURACIÓN -->
        <property name="show_sql">true</property>
        <property name="dialect">org.hibernate.dialect.MySQLDialect</property>
        <property name="hbm2ddl.auto">update</property>

        <!-- 3. TUS CLASES (Añadir aquí) -->
        <mapping class="com.tutorial.Videojuego"/>
        <mapping class="com.tutorial.Plataforma"/>
        
    </session-factory>
</hibernate-configuration>
